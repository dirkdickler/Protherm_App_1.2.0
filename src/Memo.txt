void EthernetClass::socketClose(uint8_t s)
{
  W5100.beginSPITransaction();
  W5100.execCmdSn(s, Sock_CLOSE);
  do
  {
    // Serial.println("Waiting...");
    delayMicroseconds(100);
  } while (W5100.readSnCR(s) != 0); // SOCK_CLOSED =0
  W5100.endSPITransaction();
}
